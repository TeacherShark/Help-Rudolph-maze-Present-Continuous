        function initKeysIcons() {
            keysIconsEl.innerHTML = '';
            for (let i = 0; i < TOTAL_KEYS; i++) {

        // Generate maze using DFS backtracker
        function generateMaze() {
            // Initialize maze with walls (1 = wall, 0 = path)
            maze = Array(MAZE_SIZE).fill().map(() => Array(MAZE_SIZE).fill(1));
            
            // Stack for DFS
            const stack = [];
            const visited = Array(MAZE_SIZE).fill().map(() => Array(MAZE_SIZE).fill(false));
            
            // Start from (1, 1)
            let currentX = 1;
            let currentY = 1;
            maze[currentY][currentX] = 0;
            visited[currentY][currentX] = true;
            stack.push({ x: currentX, y: currentY });
            
            const directions = [
                { x: 0, y: -2 }, // up
                { x: 2, y: 0 },  // right
                { x: 0, y: 2 },  // down
                { x: -2, y: 0 }  // left
            ];
            
            while (stack.length > 0) {
                const neighbors = [];
                
                for (const dir of directions) {
                    const nx = currentX + dir.x;
                    const ny = currentY + dir.y;
                    
                    if (nx > 0 && nx < MAZE_SIZE - 1 && ny > 0 && ny < MAZE_SIZE - 1 && !visited[ny][nx]) {
                        neighbors.push({ x: nx, y: ny, dir: dir });
                    }
                }
                
                if (neighbors.length > 0) {
                    const randomNeighbor = neighbors[Math.floor(Math.random() * neighbors.length)];
                    const nx = randomNeighbor.x;
                    const ny = randomNeighbor.y;
                    const dir = randomNeighbor.dir;
                    
                    // Break the wall between current and neighbor
                    maze[currentY + dir.y / 2][currentX + dir.x / 2] = 0;
                    maze[ny][nx] = 0;
                    visited[ny][nx] = true;
                    stack.push({ x: nx, y: ny });
                    currentX = nx;
                    currentY = ny;
                } else {
                    const prev = stack.pop();
                    if (prev) {
                        currentX = prev.x;
                        currentY = prev.y;
                    }
                }
            }
            
            // Ensure start and end are walkable
            maze[1][1] = 0;
            maze[MAZE_SIZE - 2][MAZE_SIZE - 2] = 0;
        }

        // Find random walkable position
        function getRandomWalkablePosition() {
            let attempts = 0;
            while (attempts < 100) {
                const x = Math.floor(Math.random() * (MAZE_SIZE - 2)) + 1;
                const y = Math.floor(Math.random() * (MAZE_SIZE - 2)) + 1;
                if (maze[y][x] === 0 && !(x === playerPos.x && y === playerPos.y)) {
                    return { x, y };
                }
                attempts++;
            }
            return { x: 1, y: 1 };
        }

        // Place keys randomly
        function placeKeys() {
            keys = [];
            for (let i = 0; i < TOTAL_KEYS; i++) {
                const pos = getRandomWalkablePosition();
                keys.push({ ...pos, collected: false });
            }
        }

        // Place enemies randomly
        function placeEnemies() {
            enemies = [];
            for (let i = 0; i < ENEMY_COUNT; i++) {
                const pos = getRandomWalkablePosition();
                enemies.push({ ...pos });
            }
        }

        // Initialize game
        function initGame() {
            gameActive = true;
            collectedKeys = 0;
            exitPos = null;
            enemies = [];
            playerPos = { x: 1, y: 1 };
            stepCount = 0;
            
            generateMaze();
            placeKeys();
            initKeysIcons();
            updateStats();
            
            if (startTime === null) {
                startTime = Date.now();
                gameInterval = setInterval(updateTimer, 100);
            }
            
            draw();
        }

        // Update timer display
        function updateTimer() {
            if (!startTime || !gameActive) return;
            
            const elapsed = Date.now() - startTime;
            const minutes = Math.floor(elapsed / 60000);
            const seconds = Math.floor((elapsed % 60000) / 1000);
            const tenths = Math.floor((elapsed % 1000) / 100);
            
            timerEl.textContent = `${minutes.toString().padStart(2, '0')}:${seconds.toString().padStart(2, '0')}.${tenths}`;
        }

        // Update stats display
        function updateStats() {
            stepsEl.textContent = stepCount;
        }

        // Draw game
        function draw() {
            if (!canvas) return;
            
            // Set canvas size
            const container = canvas.parentElement;
            const size = Math.min(container.clientWidth, container.clientHeight);
            canvas.width = size;
            canvas.height = size;
            
            const cellSize = size / MAZE_SIZE;
            
            // Clear canvas
            ctx.clearRect(0, 0, canvas.width, canvas.height);
            
            // Draw maze - no borders between cells, path in neon blue
            for (let y = 0; y < MAZE_SIZE; y++) {
                for (let x = 0; x < MAZE_SIZE; x++) {
                    if (maze[y][x] === 1) {
                        // Wall - dark gray
                        ctx.fillStyle = '#1e293b';
                        ctx.fillRect(x * cellSize, y * cellSize, cellSize, cellSize);
                    } else {
                        // Path - neon blue glow
                        ctx.fillStyle = '#06b6d4';
                        ctx.fillRect(x * cellSize, y * cellSize, cellSize, cellSize);
                        
                        // Add subtle glow effect to path
                        ctx.shadowColor = '#06b6d4';
                        ctx.shadowBlur = 8;
                        ctx.fillRect(x * cellSize, y * cellSize, cellSize, cellSize);
                        ctx.shadowBlur = 0;
                    }
                }
            }
            
            // Draw keys as present images
            for (const key of keys) {
                if (!key.collected && presentImage) {
                    ctx.drawImage(presentImage, 
                        (key.x + 0.5) * cellSize - cellSize * 0.35, 
                        (key.y + 0.5) * cellSize - cellSize * 0.35, 
                        cellSize * 0.7, 
                        cellSize * 0.7
                    );
                }
            }
            
            // Draw exit
            if (exitPos && collectedKeys === TOTAL_KEYS) {
                ctx.fillStyle = '#10b981';
                ctx.shadowColor = '#10b981';
                ctx.shadowBlur = 12;
                ctx.beginPath();
                ctx.arc(
                    (exitPos.x + 0.5) * cellSize,
                    (exitPos.y + 0.5) * cellSize,
                    cellSize * 0.4,
                    0,
                    Math.PI * 2
                );
                ctx.fill();
                ctx.shadowBlur = 0;
            }
            
            // Draw enemies as monster images
            for (const enemy of enemies) {
                if (monsterImage) {
                    ctx.drawImage(monsterImage, 
                        (enemy.x + 0.5) * cellSize - cellSize * 0.3, 
                        (enemy.y + 0.5) * cellSize - cellSize * 0.3, 
                        cellSize * 0.6, 
                        cellSize * 0.6
                    );
                }
            }
            
            // Draw player as player image
            if (playerImage) {
                ctx.drawImage(playerImage, 
                    (playerPos.x + 0.5) * cellSize - cellSize * 0.4, 
                    (playerPos.y + 0.5) * cellSize - cellSize * 0.4, 
                    cellSize * 0.8, 
                    cellSize * 0.8
                );
            }
        }

        // Move player
        function movePlayer(dx, dy) {
            if (!gameActive) return;
            
            const newX = playerPos.x + dx;
            const newY = playerPos.y + dy;
            
            if (newX >= 0 && newX < MAZE_SIZE && newY >= 0 && newY < MAZE_SIZE && maze[newY][newX] === 0) {
                playerPos.x = newX;
                playerPos.y = newY;
                stepCount++;
                updateStats();
                
                // Check for key pickup
                for (let i = 0; i < keys.length; i++) {
                    if (!keys[i].collected && keys[i].x === playerPos.x && keys[i].y === playerPos.y) {
                        showQuiz(i);
                        return;
                    }
                }
                
                // Check for exit
                if (exitPos && playerPos.x === exitPos.x && playerPos.y === exitPos.y) {
                    showVictory();
                    return;
                }
                
                // Check for enemy collision
                for (const enemy of enemies) {
                    if (playerPos.x === enemy.x && playerPos.y === enemy.y) {
                        // Move exit to new location
                        if (exitPos) {
                            exitPos = getRandomWalkablePosition();
                        }
                        break;
                    }
                }
                
                // Move enemies
                moveEnemies();
                
                draw();
            }
        }

        // Move enemies randomly
        function moveEnemies() {
            for (const enemy of enemies) {
                if (Math.random() < ENEMY_MOVE_PROBABILITY) {
                    const directions = [
                        { x: 0, y: -1 },
                        { x: 1, y: 0 },
                        { x: 0, y: 1 },
                        { x: -1, y: 0 }
                    ];
                    const randomDir = directions[Math.floor(Math.random() * directions.length)];
                    const newX = enemy.x + randomDir.x;
                    const newY = enemy.y + randomDir.y;
                    
                    if (newX >= 0 && newX < MAZE_SIZE && newY >= 0 && newY < MAZE_SIZE && maze[newY][newX] === 0) {
                        enemy.x = newX;
                        enemy.y = newY;
                    }
                }
            }
        }

        // Show quiz modal
        function showQuiz(keyIndex) {
            const question = questions[Math.floor(Math.random() * questions.length)];
            const [verb, option1, option2, option3, correctIndex] = question;
            
            questionTextEl.textContent = `What is the past tense of "${verb}"?`;
            answerButtonsEl.innerHTML = '';
            
            const options = [option1, option2, option3];
            options.forEach((option, index) => {
                const button = document.createElement('button');
                button.className = 'answer-btn';
                button.textContent = option;
                button.onclick = () => {
                    if (index + 1 === correctIndex) {
                        // Correct answer
                        keys[keyIndex].collected = true;
                        document.getElementById(`key-${collectedKeys}`).classList.add('collected');
                        collectedKeys++;
                        
                        // Check if all keys collected
                        if (collectedKeys === TOTAL_KEYS) {
                            exitPos = getRandomWalkablePosition();
                            placeEnemies();
                        }
                    } else {
                        // Wrong answer - move key to new location
                        const newPos = getRandomWalkablePosition();
                        keys[keyIndex].x = newPos.x;
                        keys[keyIndex].y = newPos.y;
                    }
                    
                    quizModal.classList.remove('active');
                    draw();
                };
                answerButtonsEl.appendChild(button);
            });
            
            quizModal.classList.add('active');
        }

        // Show victory modal
        function showVictory() {
            gameActive = false;
            clearInterval(gameInterval);
            
            finalTimeEl.textContent = timerEl.textContent;
            finalStepsEl.textContent = stepCount;
            victoryModal.classList.add('active');
        }

        // Reset game
        function resetGame() {
            clearInterval(gameInterval);
            startTime = Date.now();
            gameInterval = setInterval(updateTimer, 100);
            initGame();
        }

        // Event listeners
        document.addEventListener('keydown', (e) => {
            switch (e.key) {
                case 'ArrowUp':
                case 'w':
                case 'W':
                    movePlayer(0, -1);
                    break;
                case 'ArrowRight':
                case 'd':
                case 'D':
                    movePlayer(1, 0);
                    break;
                case 'ArrowDown':
                case 's':
                case 'S':
                    movePlayer(0, 1);
                    break;
                case 'ArrowLeft':
                case 'a':
                case 'A':
                    movePlayer(-1, 0);
                    break;
            }
        });

        document.getElementById('newMazeBtn').addEventListener('click', resetGame);
        document.getElementById('replayBtn').addEventListener('click', resetGame);
        document.getElementById('playAgainBtn').addEventListener('click', () => {
            victoryModal.classList.remove('active');
            resetGame();
        });

        // Close modals when clicking outside
        quizModal.addEventListener('click', (e) => {
            if (e.target === quizModal) {
                quizModal.classList.remove('active');
            }
        });

        victoryModal.addEventListener('click', (e) => {
            if (e.target === victoryModal) {
                victoryModal.classList.remove('active');
                resetGame();
            }
        });

        // Initialize game on load
        window.addEventListener('load', () => {
            loadImages();
            initGame();
            window.addEventListener('resize', draw);
        });
    </script>
